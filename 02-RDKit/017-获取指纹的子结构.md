# RDKFP
```py
from rdkit import Chem
from rdkit.Chem.rdFingerprintGenerator import GetRDKitFPGenerator,AdditionalOutput
from rdkit.Chem import Draw
import IPython.display
from rdkit.Chem.rdmolops import RDKFingerprint

bi = {}
fp = RDKFingerprint(mol,fpSize = 2048, bitInfo = bi)
# print(bi)
list_bits = []
legends = []
for x in fp.GetOnBits():
    for i in range(len(bi[x])):
        list_bits.append((mol, x, bi, i))
        legends.append(str(x))
fig = Draw.DrawRDKitBits(list_bits, legends = legends, molsPerRow = 4)
i = IPython.display.SVG(fig)
display(i)
```
