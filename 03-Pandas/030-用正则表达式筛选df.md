def filter_by_tail_id(df, tailid, num_c):
    """
    - df: 输入的 DataFrame。
    - tailid: 前缀，例如 'T1B3'。
    - num_c: C的个数，列表，例如[2,2,0,0]，0代表任意1-10。
    """
    # 构造固定部分和通配部分
    regex_parts = [f'C{num}' if num != 0 else r'C([1-9]|C10)' for num in num_c]
#     print(regex_parts)
    # 拼接正则表达式
    pattern = f'^{tailid}_' + ''.join(regex_parts) + r'$'
#     print(pattern)
    
    # 筛选符合条件的行
    filtered_df = df[df['Tail_ID'].str.contains(pattern, regex=True, na=False)]
    return filtered_df
